Bottom Tab Navigator Example

import { NavigationContainer } from '@react-navigation/native';
import { createBottomTabNavigator } from '@react-navigation/bottom-tabs';
import Ionicons from '@expo/vector-icons/Ionicons';

const Tab = createBottomTabNavigator();

export default function App() {
  return (
    <NavigationContainer>
      <Tab.Navigator
        screenOptions={({ route }) => ({
          tabBarIcon: ({ focused, color, size }) => {
            let iconName;

            if (route.name === 'Prayer Times') {
              iconName = focused ? 'time' : 'time-outline';
            } else if (route.name === 'Qibla') {
              iconName = focused ? 'compass' : 'compass-outline';
            } else if (route.name === 'Videos') {
              iconName = focused ? 'play-circle' : 'play-circle-outline';
            }

            return <Ionicons name={iconName} size={size} color={color} />;
          },
          tabBarActiveTintColor: '#2ecc71',
          tabBarInactiveTintColor: 'gray',
          headerShown: true,
        })}
      >
        <Tab.Screen name="Prayer Times" component={PrayerTimesScreen} />
        <Tab.Screen name="Qibla" component={QiblaScreen} />
        <Tab.Screen name="Videos" component={VideosScreen} />
      </Tab.Navigator>
    </NavigationContainer>
  );
}

⚠️ Important Limitations
MMKV is in-memory storage. Don't use it for large datasets (videos, images, etc.) as it will consume memory and slow down your app. Use it for small key-value data like settings, preferences, and cached prayer times.

When to Use AsyncStorage

  • Rapid prototyping with Expo Go (no native build required)
  • Small, simple projects where performance isn't critical
  • Teams without native development experience
  • When you need promise-based API consistency across your codebase

Video Player with Custom Controls

import { useVideoPlayer, VideoView } from 'expo-video';
import { useState, useEffect } from 'react';
import { View, TouchableOpacity, Text, Slider } from 'react-native';
import Ionicons from '@expo/vector-icons/Ionicons';

export default function CustomVideoPlayer({ videoUrl }) {
  const player = useVideoPlayer(videoUrl);
  const [isPlaying, setIsPlaying] = useState(false);
  const [currentTime, setCurrentTime] = useState(0);
  const [duration, setDuration] = useState(0);

  useEffect(() => {
    const interval = setInterval(() => {
      setCurrentTime(player.currentTime);
      setDuration(player.duration);
    }, 100);

    return () => clearInterval(interval);
  }, [player]);

  const togglePlayPause = () => {
    if (isPlaying) {
      player.pause();
    } else {
      player.play();
    }
    setIsPlaying(!isPlaying);
  };

  const seek = (value) => {
    player.seekTo(value);
  };

  return (
    <View style={{ flex: 1 }}>
      <VideoView
        style={{ width: '100%', height: 250 }}
        player={player}
        nativeControls={false}
      />

      <View style={{ padding: 20 }}>
        {/* Progress Bar */}
        <Slider
          minimumValue={0}
          maximumValue={duration}
          value={currentTime}
          onValueChange={seek}
          minimumTrackTintColor="#2ecc71"
          maximumTrackTintColor="#ccc"
        />

        {/* Play/Pause Button */}
        <TouchableOpacity onPress={togglePlayPause} style={{ alignItems: 'center' }}>
          <Ionicons
            name={isPlaying ? "pause-circle" : "play-circle"}
            size={60}
            color="#2ecc71"
          />
        </TouchableOpacity>

        {/* Time Display */}
        <Text>{formatTime(currentTime)} / {formatTime(duration)}</Text>
      </View>
    </View>
  );
}

function formatTime(seconds) {
  const mins = Math.floor(seconds / 60);
  const secs = Math.floor(seconds % 60);
  return `${mins}:${secs.toString().padStart(2, '0')}`;
}

Playlist Implementation

const videos = [
  { id: 1, title: 'How to Pray Fajr', url: 'https://...' },

        <Button title="Previous" onPress={playPrevious} disabled={currentIndex === 0} />
        <Button title="Next" onPress={playNext} disabled={currentIndex === videos.length - 1} />
      </View>
    </View>
  );
}

Getting User Location

import * as Location from 'expo-location';
import { useEffect, useState } from 'react';

export default function PrayerTimesScreen() {
  const [location, setLocation] = useState(null);
  const [errorMsg, setErrorMsg] = useState(null);

  useEffect(() => {
    (async () => {
      // Request permission
      let { status } = await Location.requestForegroundPermissionsAsync();
      if (status !== 'granted') {
        setErrorMsg('Permission to access location was denied');
        return;
      }

      // Get current position
      let location = await Location.getCurrentPositionAsync({
        accuracy: Location.Accuracy.Balanced,
      });

      setLocation(location);
    })();
  }, []);

  if (errorMsg) {
    return <Text>{errorMsg}</Text>;
  }

  if (!location) {
    return <Text>Getting location...</Text>;
  }

  return (
    <View>
      <Text>Latitude: {location.coords.latitude}</Text>
      <Text>Longitude: {location.coords.longitude}</Text>
    </View>
  );
}

Complete Prayer Times Implementation

import { Coordinates, CalculationMethod, PrayerTimes, Prayer } from 'adhan';
import * as Location from 'expo-location';
import { useEffect, useState } from 'react';

export default function PrayerTimesCalculator() {
  const [prayerTimes, setPrayerTimes] = useState(null);
  const [location, setLocation] = useState(null);

  useEffect(() => {
    (async () => {
      // Get location
      let { status } = await Location.requestForegroundPermissionsAsync();
      if (status !== 'granted') return;

      let userLocation = await Location.getCurrentPositionAsync({
        accuracy: Location.Accuracy.Balanced,
      });

      setLocation(userLocation.coords);

      // Calculate prayer times
      const coordinates = new Coordinates(
        userLocation.coords.latitude,
        userLocation.coords.longitude
      );

      const date = new Date();

      // Use appropriate calculation method for your region
      const params = CalculationMethod.NorthAmerica();

      const prayers = new PrayerTimes(coordinates, date, params);

      setPrayerTimes(prayers);
    })();
  }, []);

  if (!prayerTimes) {
    return <Text>Calculating prayer times...</Text>;
  }

  return (
    <View style={{ padding: 20 }}>
      <Text style={{ fontSize: 24, fontWeight: 'bold', marginBottom: 20 }}>
        Prayer Times
      </Text>

      <PrayerTimeRow name="Fajr" time={prayerTimes.fajr} />
      <PrayerTimeRow name="Sunrise" time={prayerTimes.sunrise} />
      <PrayerTimeRow name="Dhuhr" time={prayerTimes.dhuhr} />
      <PrayerTimeRow name="Asr" time={prayerTimes.asr} />
      <PrayerTimeRow name="Maghrib" time={prayerTimes.maghrib} />
      <PrayerTimeRow name="Isha" time={prayerTimes.isha} />

      {location && (
        <Text style={{ marginTop: 20, color: 'gray' }}>
          Location: {location.latitude.toFixed(4)}, {location.longitude.toFixed(4)}
        </Text>
      )}
    </View>
  );
}

function PrayerTimeRow({ name, time }) {
  const formatTime = (date) => {
    return date.toLocaleTimeString('en-US', {
      hour: 'numeric',
      minute: '2-digit',
      hour12: true
    });
  };

  return (
    <View style={{
      flexDirection: 'row',
      justifyContent: 'space-between',
      paddingVertical: 12,
      borderBottomWidth: 1,
      borderBottomColor: '#eee'
    }}>
      <Text style={{ fontSize: 18 }}>{name}</Text>
      <Text style={{ fontSize: 18, fontWeight: 'bold' }}>{formatTime(time)}</Text>
    </View>
  );
}

Caching Prayer Times with MMKV

import { MMKV } from 'react-native-mmkv';

const storage = new MMKV();

// Cache prayer times for the day
const cachePrayerTimes = (date, prayerTimes) => {
  const cacheKey = `prayer-times-${date.toISOString().split('T')[0]}`;
  storage.set(cacheKey, JSON.stringify({
    fajr: prayerTimes.fajr.toISOString(),
    sunrise: prayerTimes.sunrise.toISOString(),
    dhuhr: prayerTimes.dhuhr.toISOString(),
    asr: prayerTimes.asr.toISOString(),
    maghrib: prayerTimes.maghrib.toISOString(),
    isha: prayerTimes.isha.toISOString(),
  }));
};

// Retrieve cached prayer times
const getCachedPrayerTimes = (date) => {
  const cacheKey = `prayer-times-${date.toISOString().split('T')[0]}`;
  const cached = storage.getString(cacheKey);

  if (cached) {
    const parsed = JSON.parse(cached);
    return {
      fajr: new Date(parsed.fajr),
      sunrise: new Date(parsed.sunrise),
      dhuhr: new Date(parsed.dhuhr),
      asr: new Date(parsed.asr),
      maghrib: new Date(parsed.maghrib),
      isha: new Date(parsed.isha),
    };
  }

  return null;
};

Custom Responsive Hook Pattern

import { useWindowDimensions } from 'react-native';

export function useResponsive() {
  const { width, height } = useWindowDimensions();

  return {
    width,
    height,
    isSmall: width < 375,
    isMedium: width >= 375 && width < 768,
    isTablet: width >= 768,
    isLandscape: width > height,

    // Utility functions
    scale: (size) => (width / 375) * size, // base on iPhone X width
    verticalScale: (size) => (height / 812) * size, // base on iPhone X height
    moderateScale: (size, factor = 0.5) => {
      const baseScale = (width / 375) * size;
      return size + (baseScale - size) * factor;
    },
  };
}

// Usage
function MyComponent() {
  const { isTablet, scale } = useResponsive();

  return (
    <View style={{ padding: scale(20) }}>
      <Text style={{ fontSize: isTablet ? 24 : 18 }}>
        Responsive Component
      </Text>
    </View>
  );
}

Media Query Pattern

import { useWindowDimensions } from 'react-native';

export function useMediaQuery() {
  const { width } = useWindowDimensions();

  return {
    isMobile: width < 768,
    isTablet: width >= 768 && width < 1024,
    isDesktop: width >= 1024,

    // Specific breakpoints
    xs: width < 375,
    sm: width >= 375 && width < 768,
    md: width >= 768 && width < 1024,
    lg: width >= 1024,
  };
}

    width: isMobile ? '100%' : isTablet ? '48%' : '30%',

  };

  return <View style={cardStyles}>...</View>;

Complete App.js Example with Fonts & Icons

import { useFonts, Inter_400Regular, Inter_700Bold } from '@expo-google-fonts/inter';
import * as SplashScreen from 'expo-splash-screen';
import { useEffect } from 'react';
import { NavigationContainer } from '@react-navigation/native';
import { createBottomTabNavigator } from '@react-navigation/bottom-tabs';
import Ionicons from '@expo/vector-icons/Ionicons';

SplashScreen.preventAutoHideAsync();
const Tab = createBottomTabNavigator();

export default function App() {
  const [fontsLoaded] = useFonts({
    Inter_400Regular,
    Inter_700Bold,
  });

  useEffect(() => {
    if (fontsLoaded) {
      SplashScreen.hideAsync();
    }
  }, [fontsLoaded]);

  if (!fontsLoaded) {
    return null;
  }

  return (
    <NavigationContainer>
      <Tab.Navigator
        screenOptions={({ route }) => ({
          tabBarIcon: ({ focused, color, size }) => {
            let iconName;

            if (route.name === 'Prayer Times') {
              iconName = focused ? 'time' : 'time-outline';
            } else if (route.name === 'Qibla') {
              iconName = focused ? 'compass' : 'compass-outline';
            } else if (route.name === 'Videos') {
              iconName = focused ? 'play-circle' : 'play-circle-outline';
            }

            return <Ionicons name={iconName} size={size} color={color} />;
          },
          tabBarActiveTintColor: '#2ecc71',
          tabBarInactiveTintColor: 'gray',
          tabBarLabelStyle: {
            fontFamily: 'Inter_400Regular',
          },
          headerTitleStyle: {
            fontFamily: 'Inter_700Bold',
          },
        })}
      >
        <Tab.Screen name="Prayer Times" component={PrayerTimesScreen} />
        <Tab.Screen name="Qibla" component={QiblaScreen} />
        <Tab.Screen name="Videos" component={VideosScreen} />
      </Tab.Navigator>
    </NavigationContainer>
  );
}
